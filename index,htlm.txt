  <script>
    const qs = s => document.querySelector(s);
    const results = qs('#results');
    const msg = qs('#msg');

    function clearAll(){
      qs('#q').value = '';
      qs('#source').value = 'ambos';
      results.innerHTML = '';
      msg.innerHTML = '';
    }

    function slugify(t){ return (t||'').toLowerCase().normalize('NFD').replace(/[\u0300-\u036f]/g,'').replace(/[^a-z0-9]+/g,'-').replace(/^-+|-+$/g,''); }

    function buildLinks(query, source){
      const q = encodeURIComponent(query.trim());
      const links = [];
      if(source === 'ambos' || source === 'jus'){
        links.push({
          title:'Poder Judicial Mendoza - Jurisprudencia',
          url:`https://jusmendoza.gob.ar/biblioteca-judicial/jurisprudencia/`
        });
      }
      if(source === 'ambos' || source === 'tc'){
        links.push({
          title:'Tribunal de Cuentas - Búsqueda de Fallos',
          url:`https://www.tribunaldecuentas.mendoza.gov.ar/busqueda-de-fallos`
        });
      }
      return links;
    }

    function search(){
      const q = qs('#q').value.trim();
      const source = qs('#source').value;
      results.innerHTML = '';
      msg.innerHTML = '';
      if(!q){
        msg.innerHTML = '<div class="warn">Escribí una consulta para buscar.</div>';
        return;
      }

      const links = buildLinks(q, source);
      msg.innerHTML = '<div class="status">Consulta lista. Abrí el sitio oficial y pegá la búsqueda si necesitás revisar el texto exacto.</div>';

      const items = links.map(l => `
        <div class="item">
          <strong>${l.title}</strong>
          <div class="meta">Fuente oficial</div>
          <p style="margin:10px 0 0">Consulta: <em>${escapeHtml(q)}</em></p>
          <div class="btns" style="margin-top:12px">
            <a class="btn primary" href="${l.url}" target="_blank" rel="noopener">Abrir</a>
          </div>
        </div>
      `).join('');

      const resumen = `
        <div class="item">
          <strong>Resumen seguro</strong>
          <p style="margin:10px 0 0">No se generó un resumen automático porque todavía no se verificó el texto completo del fallo. Esto evita errores o inventos.</p>
        </div>
      `;

      results.innerHTML = items + resumen;
    }

    function escapeHtml(s){
      return s.replace(/[&<>"']/g, m => ({'&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;',"'":'&#39;'}[m]));
    }
  </script>
